iConnectable

Interface implemented by connectable display objects

void SetSocketFactory(iSocketFactory* factory)
void GetSocketFactory(iSocketFactory** factory)
Sets/gets the socket factory associated with a connectable object

factory

Reference to the socket factory object

void AddSocket(IPoint2d* pos,iSocket** socket)
Adds a socket to the connectable object.

pos

Location of the socket

socket

Reference to the newly created socket

void GetSocket(long idx,iSocket** socket)
Returns the socket at the specified index.

idx

Index of the socket to retreive

socket

Reference to the newly created socket

void FindSocket(IRect2d* rect,iSocket** socket)
Searches for a socket in the specified rectangle. Returns the first socket found.

rect

The search rectangle

socket

Reference to the newly created socket

long GetSocketCount()
Returns the number of sockets

void RemoveSocket(long idx)
Removes the socket at the specified index

idx

Index of the socket to be removed

void DrawSockets(CDC* pDC,iCoordinateMap* pMap)
Draws the sockets onto the supplied device context using the supplied mapping.

pDC

Pointer to a device context

pMap

Pointer to a coordinate mapping object

void Connect(long idx,iPlug* plug,DWORD* pdwCookie)
Connects a plug to a socket

idx

Index of the socket to connect to

plug

Plug object to connect to the socket

pdwCookie

Pointer to a variable that will receive a cookie that is used to identify the connection.

void Disconnect(long idx,DWORD dwCookie)
Disconnects a plug from a socket

idx

Index of the socket to disconnect from

dwCookie

Cookie that is used to identify the connection.
